    <script>
        const API_KEY = "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";
        const MODEL_ID = "google/gemini-2.0-flash-001";
        let isBusy = false;

        // TERMINAL NAVIGATION
        function enterTerminal() {
            document.getElementById('terminal-tray').classList.add('active');
            log("User entered Diagnostic Mode", "info");
        }

        function exitTerminal() {
            document.getElementById('terminal-tray').classList.remove('active');
            log("Returning to Standard AI UI", "info");
        }

        function log(msg, type = 'info') {
            const tray = document.getElementById('terminal-content');
            const entry = document.createElement('div');
            entry.className = `log-entry log-${type}`;
            entry.innerHTML = `[${new Date().toLocaleTimeString()}] ${type.toUpperCase()}: ${msg}`;
            tray.appendChild(entry);
            tray.scrollTop = tray.scrollHeight;
        }

        async function handleSend() {
            const input = document.getElementById('user-input');
            const chat = document.getElementById('chat-box');
            const btn = document.getElementById('send-btn');
            
            if(!input.value || isBusy) return;
            
            const msg = input.value;
            input.value = "";
            isBusy = true;
            btn.innerHTML = `<span class="loader"></span>`;
            
            // Add user message to UI
            chat.innerHTML += `<div class="flex justify-end"><div class="bg-cyan-500 text-black p-4 rounded-2xl rounded-tr-none text-[13px] font-semibold">${msg}</div></div>`;
            chat.scrollTop = chat.scrollHeight;

            log(`POST Request to ${MODEL_ID}`, "info");

            try {
                const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
                    method: "POST",
                    headers: { "Authorization": `Bearer ${API_KEY}`, "Content-Type": "application/json" },
                    body: JSON.stringify({ "model": MODEL_ID, "messages": [{"role": "user", "content": msg}] })
                });

                const data = await response.json();

                if (!response.ok) {
                    log(`FATAL: ${data.error?.message}`, "error");
                    // If we get an error, automatically pop up the terminal so the user sees why
                    enterTerminal();
                    throw new Error(data.error?.message);
                }

                const aiText = data.choices[0].message.content;
                log("Handshake Success: 200 OK", "success");

                // Add AI message to UI
                const aiId = "ai-" + Date.now();
                chat.innerHTML += `<div class="flex justify-start"><div id="${aiId}" class="glass p-4 rounded-2xl rounded-tl-none border border-white/5 text-[13px] text-cyan-400"></div></div>`;
                
                let i = 0;
                function type() {
                    if(i < aiText.length) {
                        document.getElementById(aiId).innerHTML += aiText.charAt(i);
                        i++; chat.scrollTop = chat.scrollHeight;
                        setTimeout(type, 10);
                    } else {
                        isBusy = false; btn.innerText = "SEND";
                    }
                }
                type();

            } catch (err) {
                isBusy = false;
                btn.innerText = "SEND";
                chat.innerHTML += `<div class="text-center text-[10px] text-red-500 font-bold uppercase tracking-widest pt-2">System Error: Check Neural Link</div>`;
            }
        }

        document.getElementById('user-input').addEventListener('keypress', (e) => { if(e.key === 'Enter') handleSend(); });
    </script>
