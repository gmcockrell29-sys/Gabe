    <script>
        const chatBox = document.getElementById('chat-box');
        const statusDot = document.getElementById('status-dot');
        const userInput = document.getElementById('user-input');
        const creditDisplay = document.getElementById('credit-count');
        const previewContainer = document.getElementById('image-preview-container');
        const previewImg = document.getElementById('image-preview');
        
        let userCredits = 10;
        let base64Image = null;
        const ADMIN_PASS = "1234";
        const API_KEY = "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";

        function updateCreditDisplay() {
            creditDisplay.innerText = userCredits;
            if (userCredits <= 0) {
                userInput.placeholder = "CREDITS DEPLETED";
                userInput.disabled = true;
                userInput.classList.add('opacity-30');
            } else {
                userInput.disabled = false;
                userInput.classList.remove('opacity-30');
                userInput.placeholder = "Analyze... (1 Credit)";
            }
        }

        function handleImageSelect(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    base64Image = e.target.result.split(',')[1];
                    previewImg.src = e.target.result;
                    previewContainer.classList.remove('hidden');
                };
                reader.readAsDataURL(file);
            }
        }

        function clearImage() {
            base64Image = null;
            previewContainer.classList.add('hidden');
            document.getElementById('file-input').value = "";
        }

        // Updated Admin Logic (Triggered by Button)
        function openAdmin() {
            const pass = prompt("TERMINAL ACCESS CODE REQUIRED:");
            if (pass === ADMIN_PASS) {
                document.getElementById('admin-panel').classList.remove('hidden');
            } else {
                alert("ACCESS DENIED: SECURITY BREACH LOGGED.");
            }
        }

        function closeAdmin() { document.getElementById('admin-panel').classList.add('hidden'); }

        function applyAdminSettings() {
            const newCredits = document.getElementById('admin-credit-input').value;
            userCredits = parseInt(newCredits);
            updateCreditDisplay();
            closeAdmin();
        }

        async function sendMessage() {
            const text = userInput.value;
            if (userCredits <= 0 || (!text && !base64Image)) return;

            userCredits -= 1;
            updateCreditDisplay();

            const userTextContent = text ? `<p class="mt-1">${text}</p>` : "";
            const userImgContent = base64Image ? `<img src="data:image/jpeg;base64,${base64Image}" class="w-64 rounded-xl mb-2 shadow-lg border border-white/20">` : "";

            chatBox.innerHTML += `<div class="flex justify-end"><div class="chat-bubble-user p-4 rounded-2xl rounded-tr-none max-w-[85%] text-sm shadow-2xl">${userImgContent} ${userTextContent}</div></div>`;
            
            userInput.value = "";
            const currentImg = base64Image;
            clearImage();
            chatBox.scrollTop = chatBox.scrollHeight;
            statusDot.classList.replace('bg-red-500', 'bg-green-500');

            try {
                const content = [{ type: "text", text: text || "Analyze this." }];
                if (currentImg) content.push({ type: "image_url", image_url: { url: `data:image/jpeg;base64,${currentImg}` } });

                const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
                    method: "POST",
                    headers: { "Authorization": `Bearer ${API_KEY}`, "Content-Type": "application/json" },
                    body: JSON.stringify({ "model": "google/gemini-2.0-flash-001", "messages": [{ "role": "user", "content": content }] })
                });
                
                const data = await response.json();
                const aiMsg = data.choices[0].message.content;
                chatBox.innerHTML += `<div class="flex justify-start"><div class="glass p-4 rounded-2xl rounded-tl-none max-w-[85%] text-sm border-white/10 shadow-lg">${aiMsg}</div></div>`;
                chatBox.scrollTop = chatBox.scrollHeight;
            } catch (err) { console.error(err); }
            statusDot.classList.replace('bg-green-500', 'bg-red-500');
        }

        userInput.addEventListener("keypress", (e) => { if(e.key === "Enter") sendMessage(); });
    </script>
