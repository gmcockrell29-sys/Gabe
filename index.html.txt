    <script>
        // Double-check this key in your OpenRouter dashboard!
        const API_KEY = "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";
        const MODEL_ID = "google/gemini-2.0-flash-001"; 
        
        let isBusy = false;

        function log(msg, type = 'info') {
            const tray = document.getElementById('terminal-content');
            const entry = document.createElement('div');
            entry.className = `log-entry log-${type}`;
            entry.innerHTML = `[${new Date().toLocaleTimeString()}] ${type.toUpperCase()}: ${msg}`;
            tray.appendChild(entry);
            tray.scrollTop = tray.scrollHeight;
        }

        function toggleTerminal() {
            document.getElementById('terminal-tray').classList.toggle('open');
        }

        log("System Booting...", "info");
        log("Validating API Key: " + API_KEY.substring(0, 12) + "...", "info");

        async function handleSend() {
            const input = document.getElementById('user-input');
            const chat = document.getElementById('chat-box');
            const btn = document.getElementById('send-btn');
            
            if(!input.value || isBusy) return;
            
            const msg = input.value;
            input.value = "";
            isBusy = true;
            btn.innerHTML = `<span class="loader"></span>`;
            
            chat.innerHTML += `<div class="flex justify-end"><div class="bg-cyan-500 text-black p-4 rounded-2xl rounded-tr-none text-xs font-bold">${msg}</div></div>`;
            chat.scrollTop = chat.scrollHeight;

            log(`Initiating Neural Link to ${MODEL_ID}`, "info");

            try {
                const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
                    method: "POST",
                    headers: { 
                        "Authorization": `Bearer ${API_KEY}`, 
                        "Content-Type": "application/json",
                        "X-Title": "Vision OS Recovery"
                    },
                    body: JSON.stringify({ "model": MODEL_ID, "messages": [{"role": "user", "content": msg}] })
                });

                const data = await response.json();

                if (!response.ok) {
                    if (data.error?.code === 401 || data.error?.message.includes("User not found")) {
                        log("CRITICAL: User Not Found. API Key is invalid or deleted.", "error");
                        throw new Error("AUTH_FAILURE: Check OpenRouter Dashboard");
                    }
                    throw new Error(data.error?.message || "Response Error");
                }

                const aiText = data.choices[0].message.content;
                log(`Handshake Success. Received ${data.usage.total_tokens} tokens`, "success");

                const aiId = "ai-" + Date.now();
                chat.innerHTML += `<div class="flex justify-start"><div id="${aiId}" class="bg-white/5 p-4 rounded-2xl rounded-tl-none border border-white/5 text-[12px] text-cyan-400"></div></div>`;
                
                let i = 0;
                function type() {
                    if(i < aiText.length) {
                        document.getElementById(aiId).innerHTML += aiText.charAt(i);
                        i++; chat.scrollTop = chat.scrollHeight;
                        setTimeout(type, 8);
                    } else {
                        isBusy = false; btn.innerText = "LINK";
                    }
                }
                type();

            } catch (err) {
                log(`RECOVERY HALTED: ${err.message}`, "error");
                isBusy = false;
                btn.innerText = "LINK";
                chat.innerHTML += `<div class="bg-red-500/10 p-3 rounded-xl text-[10px] text-red-500 font-bold uppercase text-center">Neural Link Failed: User Not Found</div>`;
            }
        }

        document.getElementById('user-input').addEventListener('keypress', (e) => { if(e.key === 'Enter') handleSend(); });
    </script>
